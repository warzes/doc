---------------------------------------------------
Литералы
---------------------------------------------------
Литерал		Тип
2			int
2u			unsigned
2l			long
2ul			unsigned long
2.0			double
2.0f		float
2.0;		long double

042			восьмеричные числа
0x42		шестнадцатеричные литералы
0b11111010	Бинарный литерал использует префикс 0b или 0B (C++14)

Для удобства цифры можно разделять апострофами (C++14)
long  b = 6'545'565'566
